<head>

<script type ="text/javascript">

	
var sayi1=prompt("Birinci sayýyý giriniz:");
var sayi2=prompt("Ýkinci sayýyý giriniz:");
var islem=prompt("Yapacaðýnýz iþlemi giriniz:");
	
switch(islem)
{
	case "toplama":
	document.write(eval(sayi1)+eval(sayi2));
	break;

	case "çýkarma":
	document.write(eval(sayi1)-eval(sayi2));
	break;

	case "çarpma":
	document.write(eval(sayi1)*eval(sayi2));
	break;

	case "bölme":
	document.write(eval(sayi1)/eval(sayi2));
	break;

	default:
	document.write("Yanlýþ seçim yaptýnýz!");
	break;
}
	

	
</script>
</head>
